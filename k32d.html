using namespace std;
void gotoxy(int x, int y)
{
COORD coord;
coord.X = x;
coord.Y = y;
SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), coord);
}
int main(int argc, char *argv[])
{
system("chcp 1250");
int x,y,D_jun;
cout<<" Podaj zmienna (kolumne ekranu) x=";
cin>>x;
cout<<" Podaj zmienna (wiersz ekranu) y=";
cin>>y;
cout<<" Podaj dlugosc linii D=";
cin>>D_jun; // zmienna z trzema literami

for (int i=1;i<=D_jun/2;i++) // pierwsza polowa linii
// dzielone przez 2 bo polowa, tak x+i-D_jun/2
// bo trzeba przesunac sie w lewo o polowe dlugosci linii
{
gotoxy(x+i-D_jun/2,y);
cout<<"-";
}
{
for (int i=1;i<=D_jun/2;i++)
{
gotoxy(x-y-D_jun/2,i);
cout<<"-";
}
}
system("PAUSE");
return EXIT_SUCCESS;
}
